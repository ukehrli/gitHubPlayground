# ------- Remove files from git&file system
git rm file1.txt
git rm -r foldername // folder
git commit -m "remove file1.txt"

# ------- Remove files from git NOT form git&file system
git rm --cached file1.txt
git commit -m "remove file1.txt"

# ------- Update remote git repo
git push origin branch_name
